broker: url: “${SOLACE_BROKER_URL}” # e.g., wss://mr2zq0g0f1.messaging.solace.cloud:443 clientUsername: “solace-cloud-client” password: “${SOLACE_BROKER_PASSWORD}” vpn: “${SOLACE_BROKER_VPN}”

LLM Service Configuration (e.g., OpenAI)

llmService: planningModel: “openai/${MODEL_NAME}” # e.g., “gpt-4o” generalModel: “openai/${MODEL_NAME}” # e.g., “gpt-4o” reportModel: “reportModel” # e.g., “gpt-4o” imageModel: “imageModel” # e.g., “dall-e-3” transcriptionModel: “transcriptionModel” # e.g., “whisper-1” llmServiceEndpoint: “http://${DATA_PLANE_LB}/bedrock-route” llmServiceApiKey: “nokey”

Kubernetes Service Configuration

service: type: LoadBalancer # Options: LoadBalancer, ClusterIP, NodePort annotations: # AWS NLB example service.beta.kubernetes.io/aws-load-balancer-type: “nlb-ip” service.beta.kubernetes.io/aws-load-balancer-scheme: internet-facing tls: enabled: false passphrase: "" # Optional: Add if your TLS key is encrypted

CONTAINER IMAGE REGISTRY

BUNDLED PERSISTENCE (PostgreSQL + SeaweedFS)

For production, consider using external managed services instead
